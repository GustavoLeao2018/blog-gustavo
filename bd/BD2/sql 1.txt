drop  function dobrar_salario(empregados)

CREATE or REPLACE FUNCTION dobrar_salario(empregados) RETURNS numeric AS $$
	SELECT $1.salario * 2 AS salario_dobrado;
$$ LANGUAGE SQL;

	FROM empregados;
